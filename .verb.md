## Usage

```js
var regex = require('{%= name %}');
('var foo = \'bar\';').match(regex());
//=> ['bar']

('var foo = "bar";').match(regex());
//=> ['"bar"'']

('var foo = "bar";\nvar bar = \'baz\'').match(regex());
//=> ['"bar"', '\'baz\'']

('foo bar ". // \' \\ . // \' \\ ." baz').match(regex());
//=> ['". // \' \\ . // \' \\ ."']
```
